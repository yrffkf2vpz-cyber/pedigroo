<?php namespace App\Services\Normalizers; class NormalizeCountryService { /** * Ország normalizálás. * Később: reg_no prefix alapján ország felismerés, * pd_countries integráció, többnyelvű támogatás. */ public function normalize(array $raw): array { $input = $raw['raw_country'] ?? null; // Minimális normalizálás $normalized = $input ? trim($input) : null; return [ 'country' => $normalized, 'country_code' => null, // később: országkód felismerés 'debug' => [ 'input' => $input, 'normalized' => $normalized, ], ]; } }
