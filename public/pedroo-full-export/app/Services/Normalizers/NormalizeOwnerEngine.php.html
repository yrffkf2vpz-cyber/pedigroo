<?php namespace App\Services\Normalize; use App\Models\PdOwner; use App\Models\PdDogOwner; use App\Models\Kennel; use Illuminate\Support\Arr; use Illuminate\Support\Str; class NormalizeOwnerEngine { /** * A nyers kutyaadatokb’┐Įl elo’┐Įll’┐Įtja a tulajdonosi strukt’┐Įr’┐Įt. * * @param array $rawDogData A WebImporter / NormalizePipeline ’┐Įltal ’┐Įsszerakott nyers adat. * @return array [ * 'owners' => [ PdDogOwner::create()-hez haszn’┐Įlhat’┐Į t’┐Įmb’┐Įk ... ], * ] */ public function normalize(array $rawDogData): array { $country = $rawDogData['country'] ?? null; // 1) kennel owner / breeder felold’┐Įsa $kennelOwner = $this->resolveKennelOwner($rawDogData, $country); // 2) explicit owner (ha van) $explicitOwner = $this->resolveExplicitOwner($rawDogData, $country); // 3) co-owners (ha vannak) $coOwners = $this->resolveCoOwners($rawDogData, $country); // 4) holder (aki tartja, de nem tulajdonos) $holder = $this->resolveHolder($rawDogData, $country); $ownersPivot = []; // --- PRIMARY OWNER LOGIKA --- if ($explicitOwner) { // ha van explicit owner ? o a primary $ownersPivot[] = [ 'owner_id' => $explicitOwner->id, 'ownership_type' => 'primary', 'can_edit' => true, 'acquired_at' => $rawDogData['sold_at'] ?? null, 'released_at' => null, ]; // a kennel owner ilyenkor m’┐Įr nem primary, de opcion’┐Įlisan // megjelenhet co-ownerk’┐Įnt, ha ezt k’┐Įsobb szeretn’┐Įd } else { // ha nincs explicit owner ? kennel owner = primary if ($kennelOwner) { $ownersPivot[] = [ 'owner_id' => $kennelOwner->id, 'ownership_type' => 'primary', 'can_edit' => true, 'acquired_at' => $rawDogData['birth_date_normalized'] ?? null, 'released_at' => null, ]; } } // --- CO-OWNERS --- foreach ($coOwners as $coOwner) { $ownersPivot[] = [ 'owner_id' => $coOwner->id, 'ownership_type' => 'co-owner', // itt j’┐Įn be az elved: k’┐Įrhetik, hogy b’┐Įrmelyik tulaj szerkeszthessen 'can_edit' => $rawDogData['co_owner_can_edit'] ?? false, 'acquired_at' => $rawDogData['co_owner_since'] ?? null, 'released_at' => null, ]; } // --- HOLDER --- if ($holder) { $ownersPivot[] = [ 'owner_id' => $holder->id, 'ownership_type' => 'holder', 'can_edit' => false, // holder soha nem szerkeszthet 'acquired_at' => $rawDogData['holder_since'] ?? null, 'released_at' => null, ]; } return [ 'owners' => $ownersPivot, ]; } /** * Kennel owner / breeder felold’┐Įsa. */ protected function resolveKennelOwner(array $rawDogData, ?string $country): ?PdOwner { // ha van kennel_id, megpr’┐Įb’┐Įljuk onnan if (!empty($rawDogData['kennel_id'])) { $kennel = Kennel::find($rawDogData['kennel_id']); if ($kennel && !empty($kennel->owner_name)) { return $this->findOrCreateOwner($kennel->owner_name, $country); } } // ha van explicit kennel_owner_name a raw adatban if (!empty($rawDogData['kennel_owner_name'])) { return $this->findOrCreateOwner($rawDogData['kennel_owner_name'], $country); } // fallback: ha van kennel_name, azt haszn’┐Įljuk owner n’┐Įvnek if (!empty($rawDogData['kennel_name_normalized'])) { return $this->findOrCreateOwner($rawDogData['kennel_name_normalized'], $country); } return null; } /** * Explicit owner felold’┐Įsa (ha a forr’┐Įsoldal megadja). */ protected function resolveExplicitOwner(array $rawDogData, ?string $country): ?PdOwner { $name = $rawDogData['owner_name'] ?? null; if (empty($name)) { return null; } return $this->findOrCreateOwner($name, $country); } /** * Co-owners felold’┐Įsa (pl. vesszovel elv’┐Įlasztott lista). */ protected function resolveCoOwners(array $rawDogData, ?string $country): array { $result = []; $raw = $rawDogData['co_owner_names'] ?? null; if (empty($raw)) { return $result; } $names = array_filter(array_map('trim', explode(',', $raw))); foreach ($names as $name) { if ($name === '') { continue; } $result[] = $this->findOrCreateOwner($name, $country); } return $result; } /** * Holder felold’┐Įsa (aki tartja, de nem tulajdonos). */ protected function resolveHolder(array $rawDogData, ?string $country): ?PdOwner { $name = $rawDogData['holder_name'] ?? null; if (empty($name)) { return null; } return $this->findOrCreateOwner($name, $country); } /** * Owner keres’┐Įse / l’┐Įtrehoz’┐Įsa a pd_owners t’┐Įbl’┐Įban. */ protected function findOrCreateOwner(string $name, ?string $country): PdOwner { $name = trim($name); return PdOwner::firstOrCreate( [ 'name' => $name, 'country' => $country, ], [ 'name' => $name, 'country' => $country, ] ); } }
