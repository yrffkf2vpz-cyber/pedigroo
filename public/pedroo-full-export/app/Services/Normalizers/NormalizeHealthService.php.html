<?php namespace App\Services\Normalizers; class NormalizeHealthService { public function normalize(array $rawHealth, bool $debug = false): array { $items = []; $debugLines = []; foreach ($rawHealth as $line) { $original = trim($line); $parsed = [ 'raw' => $original, 'type' => null, 'value' => null, 'lab' => null, 'date' => null, ]; // ----------------------------- // HD (Hip Dysplasia) // ----------------------------- if (preg_match('/\bHD\b/i', $original)) { $parsed['type'] = 'HD'; // Value: A, B, C, D, E, OFA Good, etc. if (preg_match('/HD\s*([A-F])\b/i', $original, $m)) { $parsed['value'] = strtoupper($m[1]); } // Lab: BVA, FCI, OFA if (preg_match('/\b(BVA|FCI|OFA)\b/i', $original, $m)) { $parsed['lab'] = strtoupper($m[1]); } } // ----------------------------- // ED (Elbow Dysplasia) // ----------------------------- if (preg_match('/\bED\b/i', $original)) { $parsed['type'] = 'ED'; // Value: 0/0, 0/1, 1/1, etc. if (preg_match('/ED\s*([0-3]\/[0-3])/i', $original, $m)) { $parsed['value'] = $m[1]; } } // ----------------------------- // DM (Degenerative Myelopathy) // ----------------------------- if (preg_match('/\bDM\b/i', $original)) { $parsed['type'] = 'DM'; // Canonical values $map = [ 'N\/N' => 'CLEAR', 'N\/DM' => 'CARRIER', 'DM\/DM' => 'AFFECTED', 'CLEAR' => 'CLEAR', 'CARRIER' => 'CARRIER', 'AFFECTED' => 'AFFECTED', ]; foreach ($map as $pattern => $canonical) { if (preg_match("/$pattern/i", $original)) { $parsed['value'] = $canonical; break; } } } // ----------------------------- // DATE detection // ----------------------------- if (preg_match('/\b(20\d{2})\b/', $original, $m)) { $parsed['date'] = $m[1] . '-01-01'; } // ----------------------------- // Only add if type recognized // ----------------------------- if ($parsed['type']) { $items[] = $parsed; } if ($debug) { $debugLines[] = $parsed; } } return [ 'items' => $items, 'debug' => $debug ? $debugLines : null, ]; } }
