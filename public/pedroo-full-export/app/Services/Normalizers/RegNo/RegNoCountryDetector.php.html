<?php namespace App\Services\Normalizers\RegNo; use App\Services\Normalizers\RegNo\RegNoNormalizer; class RegNoCountryDetector { public function __construct( protected RegNoNormalizer $normalizer ) {} /** * Fő belépési pont: ország felismerése egy regisztrációs számból. */ public function detect(?string $rawRegNo): ?string { if (!$rawRegNo) { return null; } // 1) Normalizáljuk a regisztrációs számot $normalized = $this->normalizer->normalize($rawRegNo); // 2) Prefix alapján ország felismerése if (!empty($normalized['prefix'])) { $country = $this->detectFromPrefix($normalized['prefix']); if ($country) { return $country; } } // 3) Szám + év alapján (ritka, de előfordul) if (!empty($normalized['number']) && !empty($normalized['year'])) { $country = $this->detectFromNumberYear($normalized['number'], $normalized['year']); if ($country) { return $country; } } // 4) Fallback: régi regno_rules.php minták $country = $this->detectFromLegacyRules($rawRegNo); if ($country) { return $country; } return null; } /** * Prefix alapján ország felismerése (country_rules.php). */ private function detectFromPrefix(string $prefix): ?string { $rules = config('country_rules', []); foreach ($rules as $country => $info) { if (!empty($info['prefixes'])) { foreach ($info['prefixes'] as $p) { if (stripos($prefix, $p) === 0) { return $country; } } } } return null; } /** * Szám + év alapján ország felismerése (ritka, de néha egyedi minták vannak). */ private function detectFromNumberYear(string $number, int $year): ?string { $rules = config('regno_rules', []); foreach ($rules as $country => $data) { if (!empty($data['number_year_ranges'])) { foreach ($data['number_year_ranges'] as $range) { if ( $year >= ($range['year_start'] ?? 0) && $year <= ($range['year_end'] ?? 9999) && $number >= ($range['num_start'] ?? 0) && $number <= ($range['num_end'] ?? 999999) ) { return $country; } } } } return null; } /** * Fallback: régi regno_rules.php regex minták. */ private function detectFromLegacyRules(string $raw): ?string { $rules = config('regno_rules', []); foreach ($rules as $country => $data) { if (empty($data['patterns'])) { continue; } foreach ($data['patterns'] as $pattern => $issuer) { if (@preg_match($pattern, $raw)) { return $country; } } } return null; } }
