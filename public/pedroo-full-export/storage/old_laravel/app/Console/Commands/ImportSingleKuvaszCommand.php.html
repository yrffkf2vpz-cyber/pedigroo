<?php namespace App\Console\Commands; use Illuminate\Console\Command; use App\Services\Dog\KuvaszAdatbazisDogFinder; use App\Services\Dog\DogActivationService; class ImportSingleKuvaszCommand extends Command { protected $signature = 'pedroo:import-kuvasz {name}'; protected $description = 'Importál egy kuvasz kutyát név alapján a kuvaszadazbazis.hu-ról'; public function handle( KuvaszAdatbazisDogFinder $finder, DogActivationService $activation ) { $name = $this->argument('name'); $raw = $finder->findByName($name); if (! $raw) { $this->error("Nem található kutya ezzel a névvel: {$name}"); return Command::FAILURE; } $dog = $activation->activate($raw); $this->info("Importálva: {$dog->dog_id} – {$dog->name}"); return Command::SUCCESS; } }
