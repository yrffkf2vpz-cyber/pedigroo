<?php namespace App\Console\Commands; use Illuminate\Console\Command; use App\Services\PedrooDogFetcher; class PedrooFetchOneDog extends Command { protected $signature = 'pedroo:fetch-one-dog'; protected $description = 'Fetch one dog from internet and store in pedroo_dogs'; public function handle(PedrooDogFetcher $fetcher) { $result = $fetcher->processOneDog(); if (! $result) { $this->info('No more dogs to process.'); return Command::SUCCESS; } $this->info('Processed dog ID: ' . $result->source_dog_id . ' | confidence: ' . $result->confidence); return Command::SUCCESS; } }
