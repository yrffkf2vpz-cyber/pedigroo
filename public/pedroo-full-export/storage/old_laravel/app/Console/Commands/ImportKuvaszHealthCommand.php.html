<?php namespace App\Console\Commands; use Illuminate\Console\Command; use App\Services\Health\KuvaszAdatbazisHealthNormalizer; use App\Services\Health\HealthImportService; class ImportKuvaszHealthCommand extends Command { protected $signature = 'pedroo:import-kuvasz-health {dog_id}'; protected $description = 'HealthNormalizer v2.0 futtatása egy kuvasz kutyára'; public function handle( KuvaszAdatbazisHealthNormalizer $normalizer, HealthImportService $importer ) { $dogId = $this->argument('dog_id'); $records = $normalizer->normalize($dogId); $importer->import($dogId, $records); $this->info("Health adatok importálva: {$dogId}"); } }
