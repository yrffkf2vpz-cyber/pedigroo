<?php namespace App\Services\Events; use Illuminate\Support\Facades\DB; use App\Services\Normalizers\EventNormalizer; use App\Services\Normalizers\JudgeNormalizer; use App\Services\Normalizers\RingNormalizer; use App\Services\Normalizers\DogNormalizer; class EventsIngestService { public function ingestParsed(array $parsed): void { DB::transaction(function () use ($parsed) { // 1) EVENT ? pedroo_events $eventId = EventNormalizer::id($parsed); // 2) JUDGES ? pedroo_judges foreach ($parsed['judges'] ?? [] as $judge) { JudgeNormalizer::sandboxId($judge['name']); } // 3) RINGS ? pedroo_rings foreach ($parsed['rings'] ?? [] as $ring) { RingNormalizer::id($ring['name']); } // 4) RESULTS ? pedroo_results foreach ($parsed['results'] as $row) { $dogId = DogNormalizer::id( $row['dog_name'], $row['breed'], $row['reg_no'] ?? null ); DB::table('pedroo_results')->insert([ 'event_id' => $eventId, 'dog_id' => $dogId, 'class_type' => $row['class'], 'placement' => $row['placement'], 'raw' => json_encode($row), 'hash' => sha1(json_encode($row)), 'source' => 'mksz_html', 'status' => 'pending', 'created_at' => now(), 'updated_at' => now(), ]); } }); } }
