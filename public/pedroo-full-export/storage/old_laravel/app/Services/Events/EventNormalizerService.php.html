<?php namespace App\Services\Events; use App\Services\Normalizers\JudgeNormalizer; use App\Services\Normalizers\LocationNormalizer; use App\Services\Normalizers\CountryNormalizer; use App\Services\Normalizers\BreedNormalizer; use App\Services\Normalizers\ClassNormalizer; use App\Services\Normalizers\DogNormalizer; use App\Services\Normalizers\TitleNormalizer; class EventNormalizerService { public function normalize(array $parsed): array { return [ "event_name" => $parsed["event_name"], "date" => $parsed["date"], "source" => $parsed["source"], "country_id" => CountryNormalizer::id($parsed["country"]), "location_id" => LocationNormalizer::id($parsed["city"], $parsed["location"]), "club_id" => null, // később "judges" => JudgeNormalizer::normalizeList($parsed["judges"]), "rings" => $parsed["rings"], "results" => $this->normalizeResults($parsed["results"]) ]; } protected function normalizeResults(array $results): array { $out = []; foreach ($results as $r) { $out[] = [ "dog_id" => DogNormalizer::id($r["dog_name"], $r["breed"]), "class_id" => ClassNormalizer::id($r["class"]), "placement" => intval($r["placement"]), "title_id" => TitleNormalizer::id($r["title"]), "raw_line" => $r["raw_line"] ]; } return $out; } }
