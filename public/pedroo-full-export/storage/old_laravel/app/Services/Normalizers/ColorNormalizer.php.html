<?php declare(strict_types=1); namespace App\Services\Normalizers; use Illuminate\Support\Facades\DB; class ColorNormalizer { public static function normalize(?string $input, ?int $breedId = null): ?string { if (!$input) { return null; } $clean = self::clean($input); // 1) Fajtaspecifikus színek lekérése $breedColors = $breedId ? self::getBreedColors($breedId) : []; // 2) Fajtaspecifikus pontos egyezés foreach ($breedColors as $color) { if ($clean === strtolower($color)) { return strtoupper($color); } } // 3) Fajtaspecifikus részleges egyezés foreach ($breedColors as $color) { if (str_contains($clean, strtolower($color))) { return strtoupper($color); } } // 4) Általános összetett színek if ($complex = self::matchComplex($clean)) { return $complex; } // 5) Általános egyszerű színek if ($simple = self::matchSimple($clean)) { return $simple; } // 6) Több szín kombináció if ($combo = self::matchCombination($clean)) { return $combo; } // 7) "Other color" → UNKNOWN if (str_contains($clean, 'other')) { return 'UNKNOWN'; } return 'UNKNOWN'; } private static function clean(string $input): string { $s = strtolower(trim($input)); $s = str_replace(['/', '\\', '-', '_'], ' ', $s); $s = preg_replace('/\s+/', ' ', $s); return trim($s); } private static function getBreedColors(int $breedId): array { return DB::table('pd_breed_colors') ->where('breed_id', $breedId) ->where('fci_accepted', 1) ->pluck('color_name') ->map(fn($c) => strtolower(trim($c))) ->toArray(); } private static function matchSimple(string $clean): ?string { foreach (self::simpleColors() as $standard => $variants) { foreach ($variants as $v) { if ($clean === $v) { return $standard; } } } return null; } private static function matchComplex(string $clean): ?string { foreach (self::complexColors() as $standard => $variants) { foreach ($variants as $v) { if ($clean === $v) { return $standard; } } } return null; } private static function matchCombination(string $clean): ?string { $parts = explode(' ', $clean); if (count($parts) < 2) { return null; } $normalized = []; foreach ($parts as $p) { $simple = self::matchSimple($p); if ($simple) { $normalized[] = $simple; } } if (count($normalized) >= 2) { return implode(' AND ', array_unique($normalized)); } return null; } private static function simpleColors(): array { return [ 'BLACK' => ['black'], 'WHITE' => ['white'], 'BROWN' => ['brown'], 'BLUE' => ['blue', 'gray', 'silver'], 'RED' => ['red'], 'FAWN' => ['fawn', 'beige', 'yellow'], 'BRINDLE' => ['brindle'], 'SESAME' => ['sesame'], ]; } private static function complexColors(): array { return [ 'BLACK AND TAN' => ['black and tan', 'black tan'], 'GRIZZLE AND TAN' => ['grizzle and tan', 'grizzle tan'], 'RED FAWN' => ['red fawn'], 'RED SESAME' => ['red sesame'], 'BLACK SESAME' => ['black sesame'], 'WOLF GRAY' => ['wolf gray'], 'FAWN WITH WHITE MARKINGS' => ['fawn with white markings'], ]; } }
