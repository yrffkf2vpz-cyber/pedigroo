[]; // Cache Parameters $cacheParams = array_merge($params, [ 'action' => 'get.paymentMethods', 'addons' => implode(',', $addons), ]); // Cached Query $paymentMethods = caching()->remember(PaymentMethod::class, $cacheParams, function () use ($addons, $countryCode, $sort) { $paymentMethods = PaymentMethod::query()->whereIn('name', $addons); if (!empty($countryCode)) { $countryCode = strtolower($countryCode); $findInSet = 'FIND_IN_SET("' . $countryCode . '", LOWER(countries)) > 0'; $paymentMethods->where(function ($query) use ($findInSet) { $query->whereRaw($findInSet)->orWhere(fn ($query) => $query->columnIsEmpty('countries')); }); } // Sorting $paymentMethods = $this->applySorting($paymentMethods, ['lft'], $sort); return $paymentMethods->get(); }); $resourceCollection = new EntityCollection(PaymentMethodResource::class, $paymentMethods, $params); $message = ($paymentMethods->count() <= 0) ? trans('global.no_payment_methods_found') : null; return apiResponse()->withCollection($resourceCollection, $message); } /** * Get payment method * * @param int|string $nameOrId * @param array $params * @return \Illuminate\Http\JsonResponse */ public function getEntry(int|string $nameOrId, array $params = []): JsonResponse { $identifierColumn = is_numeric($nameOrId) ? 'id' : 'name'; // Cache Parameters $cacheParams = array_merge($params, [ 'action' => 'get.paymentMethod', 'identifierColumn' => $identifierColumn, 'identifierValue' => $nameOrId, ]); $paymentMethod = caching()->remember(PaymentMethod::class, $cacheParams, function () use ($identifierColumn, $nameOrId) { $paymentMethod = PaymentMethod::query()->where($identifierColumn, $nameOrId); return $paymentMethod->first(); }); abort_if(empty($paymentMethod), 404, trans('global.payment_method_not_found')); $resource = new PaymentMethodResource($paymentMethod, $params); return apiResponse()->withResource($resource); } }
