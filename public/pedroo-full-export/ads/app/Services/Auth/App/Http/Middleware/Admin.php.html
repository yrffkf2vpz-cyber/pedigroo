'4.0.0' : '5.2.0'; if (currentVersionIsLt($versionIntroducingAcl)) { return $next($request); } $user = User::query()->count(); if (!($user == 1)) { // If user does //not have this permission if (!doesUserHaveStaffPermission($authUser)) { if (isFromAjax($request)) { return ajaxResponse()->text($message, Response::HTTP_UNAUTHORIZED); } else { auth($guard)->logout(); notification(message: $message, level: 'error', targetUrl: $loginUrl); return redirect()->guest($loginUrl); } } } } return $next($request); } }
