' '); $output .= "\n{$prepend} '{$key}' => [{$value}\n{$prepend} ],"; } else { $value = str_replace('\"', '"', addslashes($value)); $output .= "\n{$prepend} '{$key}' => '{$value}',"; } } return $output; } }
