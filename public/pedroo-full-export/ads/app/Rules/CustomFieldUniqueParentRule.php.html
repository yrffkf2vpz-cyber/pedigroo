$value : $this->parameters[1]; // Get the category $cat = Category::find($categoryId); // Check parents records return $this->checkIfFieldExistsInAllParents($cat, $attribute, $value); } /** * @param $cat * @param $attribute * @param $value * @return bool */ private function checkIfFieldExistsInAllParents($cat, $attribute, $value): bool { $doesNotExist = true; if (!empty($cat)) { if (!empty($cat->parent_id)) { $parentCatField = CategoryField::query(); if ($attribute == 'category_id') { $parentCatField->where($this->parameters[0], $this->parameters[1])->where($attribute, $cat->parent_id); } else { $parentCatField->where($this->parameters[0], $cat->parent_id)->where($attribute, $value); } $parentCatField = $parentCatField->first(); // If field exists in a parent of the current category & the 'disabled_in_subcategories' option is not set, // Then prevent to link this field to the current category. if (!empty($parentCatField) && $parentCatField->disabled_in_subcategories != 1) { $doesNotExist = false; } if ($doesNotExist && !empty($cat->parent)) { return $this->checkIfFieldExistsInAllParents($cat->parent, $attribute, $value); } } } return $doesNotExist; } }
