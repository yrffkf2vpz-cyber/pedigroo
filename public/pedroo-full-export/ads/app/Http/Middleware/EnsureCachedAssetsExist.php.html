'/*.js')) === 0) { Artisan::call('fileinput:cache-locales'); } } /** * Ensure skin cache files exist for all templates * * @return void */ private function ensureSkinsExist(): void { $frontSkinsDir = public_path('cache/css/skins/front'); $authSkinsDir = public_path('cache/css/skins/auth'); $frontSkinsExist = is_dir($frontSkinsDir) && count(glob($frontSkinsDir . '/*.css')) > 0; $authSkinsExist = is_dir($authSkinsDir) && count(glob($authSkinsDir . '/*.css')) > 0; if (!$frontSkinsExist || !$authSkinsExist) { Artisan::call('skins-css:cache'); } } /** * Ensure style CSS cache file exists * * @return void */ private function ensureStyleCssExists(): void { $cacheFile = public_path('cache/css/front-style.css'); if (!file_exists($cacheFile)) { Artisan::call('style-css:cache'); } } /** * Ensure homepage CSS cache file exists * * @return void */ private function ensureHomepageCssExists(): void { $cacheFile = public_path('cache/css/front-homepage.css'); if (!file_exists($cacheFile)) { Artisan::call('homepage-css:cache'); } } /** * Ensure add-on public assets are synced to the cache directory * * @return void */ private function ensureAddonAssetsExist(): void { // Sync assets from installed addons $this->syncAddonAssets(); // Clean up assets from unavailable addons $this->cleanupUnavailableAddonAssets(); } }
