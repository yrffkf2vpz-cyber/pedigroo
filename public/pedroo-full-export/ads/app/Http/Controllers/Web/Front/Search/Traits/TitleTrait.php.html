' rounded p-1 bg-body-tertiary'; // Title $htmlTitle = ''; // Init. $htmlTitle .= '<a href="' . urlGen()->searchWithoutQuery() . '" class="current ' . linkClass() . '">'; $htmlTitle .= '<span>' . trans('global.all_listings') . '</span>'; $htmlTitle .= '</a>'; // Location if ($isStateRequested) { // State (Admin. Division) if (!empty($state)) { $searchUrl = urlGen()->searchWithoutCity($category, $state); $htmlTitle .= ' ' . trans('global.in') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= data_get($state, 'name'); $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } } else { // City if (!empty($city)) { $searchUrl = urlGen()->searchWithoutCity($category, $city); if (config('settings.listings_list.cities_extended_searches')) { $distance = ($currentDistance == 1) ? 0 : $currentDistance; $htmlTitle .= ' ' . trans('global.within') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= trans('global.x_distance_around_city', [ 'distance' => $distance, 'unit' => getDistanceUnit(config('country.code')), 'city' => data_get($city, 'name'), ]); } else { $htmlTitle .= ' ' . trans('global.in') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= data_get($city, 'name'); } $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } } // Category if (!empty($category)) { if (!empty($parentCat)) { $searchUrl = urlGen()->searchWithoutCategory($parentCat, $city); $htmlTitle .= ' ' . trans('global.in') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= data_get($parentCat, 'name'); $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } $searchUrl = urlGen()->searchWithoutCategory($category, $city); $htmlTitle .= ' ' . trans('global.in') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= data_get($category, 'name'); $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } // Tag if (!empty($this->tag)) { $htmlTitle .= ' ' . trans('global.for') . ' '; $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . urlGen()->searchWithoutQuery() . '">'; $htmlTitle .= $this->tag; $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } // Date $postedDate = request()->input('postedDate'); $postedDateLabel = data_get($sidebar, 'periodList.' . $postedDate); if (!empty($postedDateLabel)) { $searchUrl = urlGen()->searchWithoutDate($category, $city); $htmlTitle .= trans('global.last'); $htmlTitle .= '<a rel="nofollow" class="' . $tagClass . '" href="' . $searchUrl . '">'; $htmlTitle .= $postedDateLabel; $htmlTitle .= ' <i class="bi bi-x"></i>'; $htmlTitle .= '</a>'; } view()->share('htmlTitle', $htmlTitle); return $htmlTitle; } /** * Get Breadcrumbs Tabs * * @param array|null $preSearch * @return array */ public function getBreadcrumb(?array $preSearch = []): array { // Get pre-searched objects $state = data_get($preSearch, 'admin'); $city = data_get($preSearch, 'city'); $currentDistance = data_get($preSearch, 'distance.current', 0); $category = data_get($preSearch, 'cat'); // ... $bcTab = []; // City if (!empty($city)) { $distance = ($currentDistance == 1) ? 0 : $currentDistance; $title = trans('global.in_x_distance_around_city', [ 'distance' => $distance, 'unit' => getDistanceUnit(config('country.code')), 'city' => data_get($city, 'name'), ]); $bcTab[] = collect([ 'name' => trans('global.all_listings') . ' ' . $title, 'url' => urlGen()->city($city), 'position' => !empty($category) ? 5 : 3, 'location' => true, ]); } // State (Admin. Division) if (!empty($state)) { $params = [ 'country' => config('country.icode'), 'r' => data_get($state, 'name'), ]; $searchUrl = urlGen()->search($params); $paramsToRemove = ['l', 'location', 'distance']; $searchUrl = urlBuilder($searchUrl)->removeParameters($paramsToRemove)->toString(); $title = data_get($state, 'name'); $bcTab[] = collect([ 'name' => !empty($category) ? (trans('global.all_listings') . ' ' . $title) : data_get($state, 'name'), 'url' => $searchUrl, 'position' => !empty($category) ? 5 : 3, 'location' => true, ]); } // Category $catBreadcrumb = $this->getCatBreadcrumb($category, 3); $bcTab = array_merge($bcTab, $catBreadcrumb); // Sort by Position $bcTab = array_values(Arr::sort($bcTab, function ($value) { return $value->get('position'); })); view()->share('bcTab', $bcTab); return $bcTab; } }
