$this->xPanel->setRoute(urlGen()->adminUri('report-types')); $this->xPanel->setEntityNameStrings(trans('admin.report type'), trans('admin.report types')); $this->xPanel->addButtonFromModelFunction('top', 'bulk_deletion_button', 'bulkDeletionTopButton', 'end');
