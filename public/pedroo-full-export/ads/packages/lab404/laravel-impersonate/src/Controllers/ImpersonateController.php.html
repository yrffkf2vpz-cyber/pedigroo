$this->manager->getDefaultSessionGuard(); // If the Domain Mapping addon is installed, // Then, the impersonate feature need to be disabled if (config('addons.domainmapping.installed')) { $message = trans('global.Cannot impersonate when the Domain Mapping addon is installed'); notification($message, 'error'); return redirect()->back(); } // Cannot impersonate yourself if ($id == $request->user()->getKey() && ($this->manager->getCurrentAuthGuardName() == $guardName)) { $message = trans('global.Cannot impersonate yourself'); notification($message, 'error'); return redirect()->back(); } // Cannot impersonate again if you're already impersonate a user if ($this->manager->isImpersonating()) { abort(403); } if (!$request->user()->canImpersonate()) { $message = trans('global.The current user can not impersonate'); notification($message, 'error'); return redirect()->back(); } $userToImpersonate = $this->manager->findUserById($id, $guardName); if ($userToImpersonate->canBeImpersonated()) { if ($this->manager->take($request->user(), $userToImpersonate, $guardName)) { $takeRedirect = $this->manager->getTakeRedirectTo(); if ($takeRedirect !== 'back') { return redirect()->to($takeRedirect); } } } else { $message = trans('global.The destination user can not be impersonated'); notification($message, 'error'); } return redirect()->back(); } /** * @return \Illuminate\Http\RedirectResponse */ public function leave() { if (!$this->manager->isImpersonating()) { abort(403); } $this->manager->leave(); $leaveRedirect = $this->manager->getLeaveRedirectTo(); if ($leaveRedirect !== 'back') { return redirect()->to($leaveRedirect); } return redirect()->back(); } }
