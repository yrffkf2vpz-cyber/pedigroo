(($end - $begin) + 1)); if (isset($_SERVER['HTTP_RANGE'])) { header("Content-Range: bytes $begin-$end/$size"); } header("Content-Disposition: inline; filename={$filename}"); header("Content-Transfer-Encoding: binary"); header("Last-Modified: $time"); $cur = $begin; fseek($fm, $begin, 0); while (!feof($fm) && $cur <= $end && (connection_status() == 0)) { print fread($fm, min(1024 * 16, ($end - $cur) + 1)); $cur += 1024 * 16; } } }
