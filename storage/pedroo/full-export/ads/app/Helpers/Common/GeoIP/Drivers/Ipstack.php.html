'https' : 'http'; $url = $protocol . '://api.ipstack.com/' . $ip; $query = [ 'access_key' => $accessKey, 'output' => 'json', 'language' => 'en', ]; try { $response = Http::get($url, $query); if ($response->successful()) { return $response->json(); } } catch (Throwable $e) { $response = $e; } return parseHttpRequestError($response); } }
