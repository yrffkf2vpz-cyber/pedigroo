'://ip-api.com/json/' . $ip; $query = [ 'lang' => 'en', ]; try { $response = Http::get($url, $query); if ($response->successful()) { return $response->json(); } } catch (Throwable $e) { $response = $e; } return parseHttpRequestError($response); } }
