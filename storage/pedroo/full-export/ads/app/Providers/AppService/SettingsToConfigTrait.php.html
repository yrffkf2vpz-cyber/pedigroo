[]; if (is_array($fieldValues) && count($fieldValues) > 0) { foreach ($fieldValues as $subField => $subValue) { if (!empty($subValue) || is_numeric($subValue)) { config()->set('settings.' . $setting->name . '.' . $subField, $subValue); } } } } } } catch (Throwable $e) { config()->set('settings.app.logo', config('larapen.media.logo')); } } }
