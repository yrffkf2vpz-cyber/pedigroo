(request()->header('X-VISITED-BY-SAME-SESSION') == $postId) : (session('isPostVisited') == $postId); if (!$alreadyVisited) { $this->incrementVisits($post); return true; } return false; } /** * @param $post * @return void */ private function incrementVisits($post): void { try { // Remove|unset the 'pictures' attribute (added to limit pictures number related to a selected package) $attributes = $post->getAttributes(); if (isset($attributes['pictures'])) { unset($attributes['pictures']); $post->setRawAttributes($attributes, true); } // Increment the listing's visit count $post->visits = $post->visits + 1; $post->save(); } catch (Throwable $e) { } } }
