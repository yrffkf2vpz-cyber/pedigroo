DIRECTORY_SEPARATOR . $addon->name; // Copy assets if cache doesn't exist or is outdated if (!is_dir($addonCachePath) || $this->isAddonAssetOutdated($addonPublicPath, $addonCachePath)) { // Remove old cache if it exists if (is_dir($addonCachePath)) { File::deleteDirectory($addonCachePath); } // Copy the public directory File::copyDirectory($addonPublicPath, $addonCachePath); $assetsCopied = true; } } // Update static file version if any assets were copied to bust browser cache if ($assetsCopied) { StaticFileVersion::update(); } } /** * Remove cached assets for unavailable addons */ private function cleanupUnavailableAddonAssets(): void { $cacheDir = public_path('cache/addons'); if (!is_dir($cacheDir)) { return; } $installedAddons = addon_installed_list(); $installedAddonNames = array_keys($installedAddons); // Get all cached addon directories $cachedAddonDirs = File::directories($cacheDir); foreach ($cachedAddonDirs as $cachedAddonDir) { $addonName = basename($cachedAddonDir); // Remove if addon is not installed if (!in_array($addonName, $installedAddonNames)) { File::deleteDirectory($cachedAddonDir); } } } /** * Check if cached add-on assets are outdated */ private function isAddonAssetOutdated(string $sourcePath, string $cachePath): bool { if (!is_dir($cachePath)) { return true; } // Compare modification times of source and cache directories $sourceTime = File::lastModified($sourcePath); $cacheTime = File::lastModified($cachePath); return $sourceTime > $cacheTime; } }
