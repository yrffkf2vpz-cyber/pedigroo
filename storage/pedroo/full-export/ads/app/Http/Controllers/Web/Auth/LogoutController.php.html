'-1'; // Log-out the user $data = getServiceData($this->loginService->logout($userId)); // Parsing the API response $message = data_get($data, 'message'); if (data_get($data, 'success')) { // Log out the user on a web client (Browser) logoutSession($message); // Reset Dark Mode Cookie::forget('darkTheme'); } else { $message = $message ?? trans('global.unknown_error'); flash($message)->error(); } $uriPath = property_exists($this, 'redirectAfterLogout') ? $this->redirectAfterLogout : '/'; return redirect()->to($uriPath); } }
