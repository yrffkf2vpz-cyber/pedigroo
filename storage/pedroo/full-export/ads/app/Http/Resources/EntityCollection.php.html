'\\' . $resourceClassName; } } else { $this->resourceClass = $resourceClassName; } } /** * Transform the resource into an array. * * @param \Illuminate\Http\Request $request * @param bool $unWrapped * @return array */ public function toArray(Request $request, bool $unWrapped = false): array { if (empty($this->collection) || !($this->collection instanceof Collection)) { return []; } $collection = $this->collection->transform(function ($resource) { return new $this->resourceClass($resource, $this->params); }); if ($unWrapped) { return $collection->toArray(); } return [ 'data' => $collection, ]; } }
