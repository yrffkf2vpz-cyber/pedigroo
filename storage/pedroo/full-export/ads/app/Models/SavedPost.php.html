use Illuminate\Database\Eloquent\Attributes\ObservedBy; use Illuminate\Database\Eloquent\Casts\Attribute; use Illuminate\Database\Eloquent\Relations\BelongsTo; // #[ObservedBy([SavedPostObserver::class])] class SavedPost extends BaseModel { use Crud, AppendsTrait;
