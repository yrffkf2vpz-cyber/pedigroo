} /** * Applies a specific preset configuration to the given setting. * * Determines the matching Preset class from the setting's FQN, retrieves * the preset name from the request input, and invokes the corresponding * preset method to modify or enrich the given values. * * @param string $settingClassFQN * @param array $value * @return array */ protected static function applyPreset(string $settingClassFQN, array $value = []): array { $settingClassName = class_basename($settingClassFQN);
