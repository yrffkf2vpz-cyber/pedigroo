return 'dummy' } } describe('Base Component', () => { let fixtureEl const name = 'dummy' let element let instance const createInstance = () => { fixtureEl.innerHTML = '<div id="foo"></div>' element = expect(DummyClass.DATA_KEY).toEqual(`bs.${name}`) }) }) describe('NAME', () => { it('should throw an Error if it is not initialized', () => { expect(() => { // eslint-disable-next-line no-unused-expressions BaseComponent.NAME }).toThrowError(Error) }) it('should return plugin NAME', () => { expect(DummyClass.NAME).toEqual(name) }) }) describe('EVENT_KEY', () => { it('should return }) }) describe('getInstance', () => { it('should return an instance', () => { createInstance() expect(DummyClass.getInstance(element)).toEqual(instance)
