null, 'country_id' => $validated['country_id'] ?? null, 'source' => $validated['source'] ?? null, 'external_id' => $validated['external_id'] ?? null, 'updated_at' => now(), ]); return response()->json([ 'status' => 'updated', 'id' => $existing, ]); } // INSERT $id = DB::table('pd_championships')->insertGetId([ 'dog_id' => $validated['dog_id'], 'event_id' => $validated['event_id'] ?? null, 'title_definition_id'=> $validated['title_definition_id'], 'country_id' => $validated['country_id'] ?? null, 'date' => $validated['date'] ?? null, 'source' => $validated['source'] ?? null, 'external_id' => $validated['external_id'] ??
