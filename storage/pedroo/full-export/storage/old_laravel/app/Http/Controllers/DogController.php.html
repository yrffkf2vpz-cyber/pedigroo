<?php namespace App\Http\Controllers; use Illuminate\Http\Request; use Illuminate\Support\Facades\DB; class DogController extends Controller { public function search(Request $request) { $q = $request->input('q'); if (!$q) { return response()->json([]); } $dogs = DB::table('pd_dogs') ->where('name', 'like', "%{$q}%") ->limit(50) ->get(); return response()->json($dogs); } public function show($id) { $dog = DB::table('pd_dogs')->where('id', $id)->first(); if (!$dog) { return response()->json(['error' => 'Not found'], 404); } return response()->json($dog); } }
