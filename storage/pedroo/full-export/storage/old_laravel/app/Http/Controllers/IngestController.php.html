<?php namespace App\Http\Controllers; use Illuminate\Http\Request; use App\Services\Ingest\IngestExcelService; use App\Services\Ingest\IngestApiService; use App\Services\Ingest\IngestScraperService; use App\Services\Ingest\IngestPdfService; class IngestController extends Controller { public function excel(Request $request, IngestExcelService $excel) { $file = $request->file('file'); if (!$file) { return response()->json(['error' => 'No file uploaded'], 422); } $excel->import($file->getRealPath()); return response()->json(['status' => 'ok']); } public function api(Request $request, IngestApiService $api) { $dogs = $request->input('dogs', []); $api->import($dogs); return response()->json(['status' => 'ok']); } public function scraper(Request $request, IngestScraperService $scraper) { $dogs = $request->input('dogs', []); $scraper->import($dogs); return response()->json(['status' => 'ok']); } public function pdf(Request $request, IngestPdfService $pdf) { $file = $request->file('file'); if (!$file) { return response()->json(['error' => 'No file uploaded'], 422); } $pdf->import($file->getRealPath()); return response()->json(['status' => 'ok']); } }
