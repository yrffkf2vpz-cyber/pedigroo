<?php namespace App\Console\Commands; use Illuminate\Console\Command; use App\Services\Ingest\IngestExcelService; class PedrooIngestCommand extends Command { protected $signature = 'pedroo:ingest {file}'; protected $description = 'Ingest dogs from an Excel file into Pedroo'; public function handle(IngestExcelService $excel) { $filePath = $this->argument('file'); $this->info("Starting ingest from: {$filePath}"); $excel->import($filePath); $this->info("Ingest completed."); } }
