null, 'params' => json_encode($params, JSON_UNESCAPED_UNICODE), 'value_before' => $event['value_before'] ?? null, 'value_after' => $event['value_after'] ?? null, 'meta' => isset($event['meta']) ? json_encode($event['meta'], JSON_UNESCAPED_UNICODE) : null, 'created_at' => now(), 'updated_at' => now(), ]); } /** * Generate params JSON automatically based on event type. * * @param array $event * @return array */ protected function generateParams(array $event): array { $params = []; switch ($event['type']) { case 'country_code_change': $params = [ 'country' => $event['country'] ??
