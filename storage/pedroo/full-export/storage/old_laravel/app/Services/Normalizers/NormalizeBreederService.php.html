<?php namespace App\Services\Normalizers; class NormalizeBreederService { /** * Breeder normalizálás. * Később: duplikációkezelés, országfelismerés, kennel-kapcsolat. */ public function normalize(array $raw): array { $input = $raw['raw_breeder'] ?? null; // Minimális normalizálás: trim + üres string kezelése $normalized = $input ? trim($input) : null; return [ 'breeder_name' => $normalized, 'breeder_id' => null, // később: automatikus breeder matching 'debug' => [ 'input' => $input, 'normalized' => $normalized, ], ]; } }
