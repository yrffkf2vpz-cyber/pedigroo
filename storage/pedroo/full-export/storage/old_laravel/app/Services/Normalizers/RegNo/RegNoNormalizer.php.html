<?php namespace App\Services\Normalizers\RegNo; class RegNoNormalizer { public function normalize(?string $raw): ?array { if (!$raw) { return null; } $raw = trim($raw); $country = CountryPatterns::detectCountry($raw); return match ($country) { 'HU' => $this->normalizeHU($raw), 'UK' => $this->normalizeUK($raw), 'US' => $this->normalizeUS($raw), 'FR' => $this->normalizeFR($raw), 'DE' => $this->normalizeDE($raw), default => $this->normalizeGeneric($raw), }; } protected function normalizeHU(string $raw): array { $r = strtoupper(preg_replace('/\s+/', ' ', $raw)); // MET 12345/2020 if (preg_match('/MET\s+(\d{1,6})\/(\d{4})/', $r, $m)) { return [ 'raw' => $raw, 'normalized' => "MET {$m[1]}/{$m[2]}", 'country' => 'HU', 'prefix' => 'MET', 'number' => $m[1], 'year' => $m[2], 'confidence' => 0.95, ]; } return $this->normalizeGeneric($raw, 'HU'); } protected function normalizeUK(string $raw): array { $r = strtoupper(preg_replace('/\s+/', ' ', $raw)); // KC AT01234506 if (preg_match('/(KC\s+)?([A-Z]{2}\d{6}\d{2})/', $r, $m)) { $num = $m[2]; return [ 'raw' => $raw, 'normalized' => "KC {$num}", 'country' => 'UK', 'prefix' => 'KC', 'number' => $num, 'year' => null, 'confidence' => 0.9, ]; } return $this->normalizeGeneric($raw, 'UK'); } protected function normalizeUS(string $raw): array { $r = strtoupper(preg_replace('/\s+/', ' ', $raw)); // AKC DN12345607 if (preg_match('/(AKC\s+)?([A-Z]{2}\d{8})/', $r, $m)) { $num = $m[2]; return [ 'raw' => $raw, 'normalized' => "AKC {$num}", 'country' => 'US', 'prefix' => 'AKC', 'number' => $num, 'year' => null, 'confidence' => 0.9, ]; } return $this->normalizeGeneric($raw, 'US'); } protected function normalizeFR(string $raw): array { $r = strtoupper(preg_replace('/\s+/', ' ', $raw)); // LOF 123456/01234 if (preg_match('/LOF\s+(\d{1,6})\/(\d{1,5})/', $r, $m)) { return [ 'raw' => $raw, 'normalized' => "LOF {$m[1]}/{$m[2]}", 'country' => 'FR', 'prefix' => 'LOF', 'number' => $m[1], 'year' => null, 'confidence' => 0.9, ]; } return $this->normalizeGeneric($raw, 'FR'); } protected function normalizeDE(string $raw): array { $r = strtoupper(preg_replace('/\s+/', ' ', $raw)); // VDH 123456 if (preg_match('/VDH\s+(\d{4,8})/', $r, $m)) { return [ 'raw' => $raw, 'normalized' => "VDH {$m[1]}", 'country' => 'DE', 'prefix' => 'VDH', 'number' => $m[1], 'year' => null, 'confidence' => 0.85, ]; } return $this->normalizeGeneric($raw, 'DE'); } protected function normalizeGeneric(string $raw, ?string $country = null): array { return [ 'raw' => $raw, 'normalized' => trim(preg_replace('/\s+/', ' ', $raw)), 'country' => $country, 'prefix' => null, 'number' => null, 'year' => null, 'confidence' => 0.3, ]; } }
