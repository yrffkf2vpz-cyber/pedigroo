<?php namespace App\Services\Normalizers; use App\Services\Ingest\DogRecordBuilder; // Normalize modules use App\Services\Normalizers\NormalizeDogService; use App\Services\Normalizers\NormalizeParentService; use App\Services\Normalizers\NormalizeResultsService; use App\Services\Normalizers\NormalizeHealthService; use App\Services\Normalizers\NormalizeBreederService; use App\Services\Normalizers\NormalizeOwnerService; use App\Services\Normalizers\NormalizeKennelService; use App\Services\Normalizers\NormalizeColorService; use App\Services\Normalizers\NormalizeBreedService; use App\Services\Normalizers\NormalizeCountryService; use App\Services\Normalizers\RegNo\RegNoService; use App\Services\Normalizers\Name\NameParser; // Promotion modules use App\Services\Normalizers\PromotionNormalizer; use App\Services\Normalizers\TitleNormalizer; use App\Services\Normalizers\ChampionPointCalculator; // Archive modules use App\Services\Normalizers\HistoryWriter; use App\Services\Normalizers\ChangeDetector; use App\Services\Normalizers\DuplicateResolver; class NormalizePipelineService { public function __construct( // INGEST protected DogRecordBuilder $builder, // NORMALIZE protected NormalizeDogService $normalizeDog, protected NormalizeParentService $normalizeParents, protected NormalizeResultsService $normalizeResults, protected ParentMatchingService $parentMatcher, protected NormalizeHealthService $normalizeHealth, protected NormalizeBreederService $normalizeBreeder, protected NormalizeOwnerService $normalizeOwner, protected NormalizeKennelService $normalizeKennel, protected NormalizeColorService $normalizeColor, protected NormalizeBreedService $normalizeBreed, protected NormalizeCountryService $normalizeCountry, protected RegNoService $regNoService, // PROMOTION protected PromotionNormalizer $promotionNormalizer, protected TitleNormalizer $titleNormalizer, protected ChampionPointCalculator $championPoints, // ARCHIVE protected HistoryWriter $historyWriter, protected ChangeDetector $changeDetector, protected DuplicateResolver $duplicateResolver, ) {} /** * Teljes ingest → normalize → promote → archive pipeline. */ public function process(array $rawDogRecord, bool $debug = false): array { // --------------------------------------------------------- // 1) INGEST – egységes raw rekord // --------------------------------------------------------- $raw = $this->builder->fromExcelRow($rawDogRecord); // --------------------------------------------------------- // 2) NORMALIZE – minden modul külön // --------------------------------------------------------- $dog = $this->normalizeDog->normalize($raw, $debug); // PARENTS – normalizálás $father = $this->normalizeParents->normalize($raw['raw_father'] ?? null); $mother = $this->normalizeParents->normalize($raw['raw_mother'] ?? null); // PARENTS – matching $fatherMatch = $this->parentMatcher->match($father); $motherMatch = $this->parentMatcher->match($mother); // DOG rekordba beírjuk a szülő ID-ket $dog['father_id'] = $fatherMatch['matched_id'] ?? null; $dog['mother_id'] = $motherMatch['matched_id'] ?? null; // További normalizálások $results = $this->normalizeResults->normalize($raw); $health = $this->normalizeHealth->normalize($raw['health'] ?? [], $debug); $breeder = $this->normalizeBreeder->normalize($raw); $owner = $this->normalizeOwner->normalize($raw); $kennel = $this->normalizeKennel->normalize($raw); $color = $this->normalizeColor->normalize($raw); $breed = $this->normalizeBreed->normalize($raw); $country = $this->normalizeCountry->normalize($raw); $regno = $this->regNoService->process($raw['raw_reg_no'] ?? null); // --------------------------------------------------------- // SZÍNEK ÁTVÉTELE A DOG REKORDBA // --------------------------------------------------------- $dog['color'] = $color['color'] ?? null; $dog['official_color'] = $color['official_color'] ?? null; $dog['birth_color'] = $color['birth_color'] ?? null; // --------------------------------------------------------- // 3) PROMOTION – címek, pontok, rangok // --------------------------------------------------------- $promotions = $this->promotionNormalizer->normalize($results); $titles = $this->titleNormalizer->normalize($promotions); $points = $this->championPoints->calculate($results); // --------------------------------------------------------- // 4) ARCHIVE – történet, változás, duplikáció // --------------------------------------------------------- $parents = [ 'father' => $father, 'mother' => $mother, ]; $history = $this->historyWriter->write($dog, $parents, $results); $changes = $this->changeDetector->detect($dog); $duplicates = $this->duplicateResolver->resolve($dog); // --------------------------------------------------------- // 5) VÉGSŐ ÖSSZEÁLLÍTOTT REKORD + DEBUG // --------------------------------------------------------- return [ 'dog' => $dog, 'parents' => $parents, 'parent_matches' => [ 'father' => $fatherMatch, 'mother' => $motherMatch, ], 'results' => $results, 'health' => $health, 'breeder' => $breeder, 'owner' => $owner, 'kennel' => $kennel, 'color' => $color, 'breed' => $breed, 'country' => $country, 'regno' => $regno, 'promotions' => $promotions, 'titles' => $titles, 'points' => $points, 'history' => $history, 'changes' => $changes, 'duplicates' => $duplicates, 'debug' => $debug ? [ 'ingest' => [ 'raw_input' => $rawDogRecord, 'builder_output' => $raw, ], 'normalize' => [ 'dog' => $dog['debug'] ?? null, 'parents' => [ 'father' => $father['debug'] ?? null, 'mother' => $mother['debug'] ?? null, ], 'parent_matches' => [ 'father' => $fatherMatch['debug'] ?? null, 'mother' => $motherMatch['debug'] ?? null, ], 'results' => $results['debug'] ?? null, 'health' => $health['debug'] ?? null, 'breeder' => $breeder['debug'] ?? null, 'owner' => $owner['debug'] ?? null, 'kennel' => $kennel['debug'] ?? null, 'color' => $color['debug'] ?? null, 'breed' => $breed['debug'] ?? null, 'country' => $country['debug'] ?? null, 'regno' => $regno['debug'] ?? null, ], 'promotion' => [ 'promotions' => $promotions['debug'] ?? null, 'titles' => $titles['debug'] ?? null, 'points' => $points['debug'] ?? null, ], 'archive' => [ 'history' => $history['debug'] ?? null, 'changes' => $changes['debug'] ?? null, 'duplicates' => $duplicates['debug'] ?? null, ], ] : null, ]; } }
