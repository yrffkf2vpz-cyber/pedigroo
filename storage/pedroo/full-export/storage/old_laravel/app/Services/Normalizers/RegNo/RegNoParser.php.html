<?php namespace App\Services\Normalizers\RegNo; use App\Services\Normalizers\RegNo\RegNoNormalizer; use App\Services\Normalizers\RegNo\RegNoCountryDetector; use App\Services\Normalizers\RegNo\RegNoParser; class RegNoParser { protected RegNoNormalizer $normalizer; protected RegNoCountryDetector $countryDetector; public function __construct( RegNoNormalizer $normalizer, RegNoCountryDetector $countryDetector ) { $this->normalizer = $normalizer; $this->countryDetector = $countryDetector; } /** * Fő belépési pont: teljes regisztrációs szám feldolgozása. */ public function parse(?string $raw): array { if (!$raw) { return $this->emptyResult(); } // 1) Normalizálás (prefix, szám, év) $normalized = $this->normalizer->normalize($raw); // 2) Ország felismerése $country = $this->countryDetector->detect($raw); // 3) Issuer felismerése (regno_rules.php alapján) $issuer = $this->detectIssuer($raw); return [ 'raw' => $raw, 'prefix' => $normalized['prefix'], 'number' => $normalized['number'], 'year' => $normalized['year'], 'country' => $country, 'issuer' => $issuer, 'clean' => $normalized['clean'], ]; } /** * Issuer felismerése regno_rules.php alapján. */ private function detectIssuer(string $raw): ?string { $rules = config('regno_rules', []); foreach ($rules as $country => $data) { if (empty($data['patterns'])) { continue; } foreach ($data['patterns'] as $pattern => $issuer) { if (@preg_match($pattern, $raw)) { return $issuer; } } } return null; } /** * Üres eredmény sablon. */ private function emptyResult(): array { return [ 'raw' => null, 'prefix' => null, 'number' => null, 'year' => null, 'country' => null, 'issuer' => null, 'clean' => null, ]; } }
