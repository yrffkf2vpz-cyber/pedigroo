<?php namespace App\Services\Normalizers; class NormalizeKennelService { /** * Kennel normalizálás. * Később: kennel matching, országfelismerés, kennel prefix/suffix logika. */ public function normalize(array $raw): array { $input = $raw['raw_kennel'] ?? null; // Minimális normalizálás $normalized = $input ? trim($input) : null; return [ 'kennel_name' => $normalized, 'kennel_id' => null, // később: automatikus kennel matching 'debug' => [ 'input' => $input, 'normalized' => $normalized, ], ]; } }
