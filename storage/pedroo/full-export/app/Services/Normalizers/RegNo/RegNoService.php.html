<?php namespace App\Services\Normalizers\RegNo; use App\Services\Normalizers\RegNo\RegNoNormalizer; use App\Services\Normalizers\RegNo\RegNoCountryDetector; use App\Services\Normalizers\RegNo\RegNoParser; class RegNoService { public function __construct( protected RegNoNormalizer $normalizer, protected RegNoCountryDetector $countryDetector, protected RegNoParser $parser ) {} /** * Fő belépési pont: teljes regisztrációs szám feldolgozása + DEBUG támogatás. */ public function process(?string $rawRegNo, bool $debug = false): array { if (!$rawRegNo) { return $this->emptyResult($debug); } // --------------------------------------------------------- // 1) Normalizálás (prefix, szám, év) // --------------------------------------------------------- $normalized = $this->normalizer->normalize($rawRegNo); // --------------------------------------------------------- // 2) Ország felismerése // --------------------------------------------------------- $country = $this->countryDetector->detect($rawRegNo); // --------------------------------------------------------- // 3) Teljes parse (issuer + clean) // --------------------------------------------------------- $parsed = $this->parser->parse($rawRegNo); // --------------------------------------------------------- // 4) DEBUG STRUKTÚRA // --------------------------------------------------------- $debugData = $debug ? [ 'raw' => $rawRegNo, 'normalizer' => [ 'prefix' => $normalized['prefix'], 'number' => $normalized['number'], 'year' => $normalized['year'], ], 'country_detection' => [ 'detected_country' => $country, ], 'parser' => [ 'issuer' => $parsed['issuer'], 'clean' => $parsed['clean'], ], ] : null; // --------------------------------------------------------- // 5) VÉGSŐ VISSZATÉRÉSI STRUKTÚRA // --------------------------------------------------------- return [ 'raw' => $rawRegNo, 'prefix' => $normalized['prefix'], 'number' => $normalized['number'], 'year' => $normalized['year'], 'country' => $country, 'issuer' => $parsed['issuer'], 'clean' => $parsed['clean'], 'debug' => $debugData, ]; } private function emptyResult(bool $debug): array { return [ 'raw' => null, 'prefix' => null, 'number' => null, 'year' => null, 'country' => null, 'issuer' => null, 'clean' => null, 'debug' => $debug ? [ 'raw' => null, 'normalizer' => null, 'country_detection' => null, 'parser' => null, ] : null, ]; } }
