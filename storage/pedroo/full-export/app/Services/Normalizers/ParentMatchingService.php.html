<?php namespace App\Services\Normalizers; use App\Models\Dog; class ParentMatchingService { /** * Szülők automatikus felismerése. * Később: reg_no + név + kennel + ország alapján fuzzy matching. */ public function match(array $parent): array { // Ha nincs név és nincs reg_no → nincs mit keresni if (!$parent['name'] && !$parent['reg_no']) { return [ 'matched_id' => null, 'matched' => null, 'debug' => [ 'reason' => 'no_input', 'input' => $parent, ], ]; } // 1) REG_NO alapú pontos egyezés if ($parent['reg_no']) { $dog = Dog::where('reg_no', $parent['reg_no'])->first(); if ($dog) { return [ 'matched_id' => $dog->id, 'matched' => [ 'name' => $dog->name, 'reg_no' => $dog->reg_no, ], 'debug' => [ 'reason' => 'reg_no_match', 'input' => $parent, ], ]; } } // 2) NÉV alapú keresés (minimális skeleton) if ($parent['name']) { $dog = Dog::where('name', $parent['name'])->first(); if ($dog) { return [ 'matched_id' => $dog->id, 'matched' => [ 'name' => $dog->name, 'reg_no' => $dog->reg_no, ], 'debug' => [ 'reason' => 'name_match', 'input' => $parent, ], ]; } } // 3) Nincs találat return [ 'matched_id' => null, 'matched' => null, 'debug' => [ 'reason' => 'no_match', 'input' => $parent, ], ]; } }
