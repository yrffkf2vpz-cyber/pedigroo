<?php namespace App\Services\Normalizers; class NormalizeParentService { public function normalize(?string $rawName): array { if (!$rawName) { return [ 'name' => null, 'reg_no' => null, 'country' => null, 'debug' => [ 'input' => null, ], ]; } return [ 'name' => trim($rawName), 'reg_no' => null, 'country' => null, 'debug' => [ 'input' => $rawName, ], ]; } }
